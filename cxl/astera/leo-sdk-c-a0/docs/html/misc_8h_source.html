88static __inline__ bool bitmap_word_array_is_bitset(uint32_t *p_bmp,
89 uint32_t bit_num) {
90 uint32_t bmp_ar_idx = bit_num >> BITMAP_WORD_ARRAY_WORD_SIZE_BITS_SHIFT;
91 uint32_t bmp_bit_idx =
92 bit_num - (bmp_ar_idx << BITMAP_WORD_ARRAY_WORD_SIZE_BITS_SHIFT);
93
94 return ((p_bmp[bmp_ar_idx] & (1 << bmp_bit_idx)) >> bmp_bit_idx);
95}

104static __inline__ void bitmap_word_array_bit_set(uint32_t *p_bmp,
105 uint32_t bit_num) {
106 uint32_t bmp_ar_idx = bit_num >> BITMAP_WORD_ARRAY_WORD_SIZE_BITS_SHIFT;
107 uint32_t bmp_bit_idx =
108 bit_num - (bmp_ar_idx << BITMAP_WORD_ARRAY_WORD_SIZE_BITS_SHIFT);

110 p_bmp[bmp_ar_idx] |= (1 << bmp_bit_idx);
